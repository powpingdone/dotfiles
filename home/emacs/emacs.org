** Evil Mode (vim binds)

#+begin_src emacs-lisp

      (use-package evil
        :demand t
        :init
        (setq evil-vsplit-window-right t
        evil-split-window-below t
        evil-toggle-key "C-~"
        evil-want-keybinding t)
        :config
        (evil-mode 1))

      (use-package evil-collection
        :after evil
        :config
        (evil-collection-init))

      (use-package evil-tutor)

#+end_src
